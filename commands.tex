\newcommand{\largetitle}[1]{
    {
        \newcolumntype{B}{%
        >{\Large \centering \columncolor[gray]{.9}[.5\tabcolsep] \bfseries }%

            \medskip
            \begin{tabular}{B}
            \underline{\uppercase{#1}}
            \end{tabular}
        }

    }
}
